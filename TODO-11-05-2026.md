# 🧠 `PhysioTimelinePage.tsx`

```tsx
import React, { useState } from "react"
import { PhysioChart } from "../components/physio/PhysioChart"
import { EventInputPanel } from "../components/physio/EventInputPanel"
import { EventList } from "../components/physio/EventList"
import { SimulationControls } from "../components/physio/SimulationControls"

export type PhysioEvent = {
  id: string
  type: "coffee" | "nicotine" | "thc" | "meal"
  dose: number
  timestamp: number
}

export const PhysioTimelinePage: React.FC = () => {
  const [events, setEvents] = useState<PhysioEvent[]>([])
  const [loading, setLoading] = useState(false)
  const [data, setData] = useState<any>(null)

  const addEvent = (event: PhysioEvent) => {
    setEvents((prev) => [...prev, event])
  }

  const removeEvent = (id: string) => {
    setEvents((prev) => prev.filter((e) => e.id !== id))
  }

  const simulate = async () => {
    setLoading(true)

    const res = await fetch("/api/physio/simulate", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        events,
        context: {
          fasted: true,
          sleepDebt: 0.3,
          stressLevel: 0.5,
        },
        horizonMinutes: 180,
      }),
    })

    const json = await res.json()
    setData(json)
    setLoading(false)
  }

  return (
    <div style={styles.page}>
      <h1 style={styles.title}>Physio Timeline</h1>

      <div style={styles.grid}>
        <div style={styles.left}>
          <EventInputPanel onAdd={addEvent} />
          <EventList events={events} onRemove={removeEvent} />
          <SimulationControls onSimulate={simulate} loading={loading} />
        </div>

        <div style={styles.right}>
          <PhysioChart data={data} />
        </div>
      </div>
    </div>
  )
}

const styles: Record<string, React.CSSProperties> = {
  page: {
    padding: 20,
    fontFamily: "sans-serif",
    background: "#0b0f14",
    color: "#fff",
    minHeight: "100vh",
  },
  title: {
    marginBottom: 20,
  },
  grid: {
    display: "grid",
    gridTemplateColumns: "320px 1fr",
    gap: 20,
  },
  left: {
    display: "flex",
    flexDirection: "column",
    gap: 12,
  },
  right: {
    background: "#111826",
    borderRadius: 12,
    padding: 12,
    height: "70vh",
  },
}
```

# 📊 `PhysioChart.tsx` (Nivo Line Chart)

```tsx
import React from "react"
import { ResponsiveLine } from "@nivo/line"

type Props = {
  data: any
}

export const PhysioChart: React.FC<Props> = ({ data }) => {
  if (!data) {
    return (
      <div style={{ color: "#aaa" }}>
        No simulation yet. Add events and run simulation.
      </div>
    )
  }

  const chartData = [
    {
      id: "cortisol",
      data: data.curves.cortisol.map((y: number, i: number) => ({
        x: i,
        y,
      })),
    },
    {
      id: "dopamine",
      data: data.curves.dopamine.map((y: number, i: number) => ({
        x: i,
        y,
      })),
    },
    {
      id: "glucose",
      data: data.curves.glucose.map((y: number, i: number) => ({
        x: i,
        y,
      })),
    },
  ]

  return (
    <ResponsiveLine
      data={chartData}
      margin={{ top: 20, right: 20, bottom: 40, left: 40 }}
      xScale={{ type: "point" }}
      yScale={{ type: "linear", min: "auto", max: "auto" }}
      axisBottom={{
        tickRotation: -45,
      }}
      colors={{ scheme: "category10" }}
      pointSize={4}
      pointBorderWidth={1}
      useMesh={true}
      theme={{
        axis: {
          ticks: {
            text: {
              fill: "#aaa",
            },
          },
        },
        grid: {
          line: {
            stroke: "#222",
          },
        },
      }}
    />
  )
}
```

# ➕ `EventInputPanel.tsx`

```tsx
import React, { useState } from "react"

type Props = {
  onAdd: (event: any) => void
}

export const EventInputPanel: React.FC<Props> = ({ onAdd }) => {
  const [type, setType] = useState("coffee")
  const [dose, setDose] = useState(1)

  const add = () => {
    onAdd({
      id: crypto.randomUUID(),
      type,
      dose,
      timestamp: Date.now(),
    })
  }

  return (
    <div style={box}>
      <h3>Add Event</h3>

      <select value={type} onChange={(e) => setType(e.target.value)}>
        <option value="coffee">Coffee</option>
        <option value="nicotine">Nicotine</option>
        <option value="thc">THC</option>
        <option value="meal">Meal</option>
      </select>

      <input
        type="range"
        min={1}
        max={10}
        value={dose}
        onChange={(e) => setDose(Number(e.target.value))}
      />

      <button onClick={add}>Add</button>
    </div>
  )
}

const box: React.CSSProperties = {
  background: "#111826",
  padding: 12,
  borderRadius: 8,
}
```
